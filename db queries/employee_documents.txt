CREATE TABLE `employee_documents` (
  `id` int NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `employee_id` varchar(20)  NOT NULL,
  `document_category` varchar(250)  DEFAULT NULL,
  `document_name` varchar(250)  DEFAULT NULL,
  `expiry_date` date DEFAULT NULL,
  `description` text DEFAULT NULL,
  `status_ind` TINYINT(1) DEFAULT 1,
   check(status_ind IN (0,1)),
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

   FOREIGN KEY (employee_id) REFERENCES employees (id)
);

CREATE TABLE employee_documents (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    employee_id INT UNSIGNED NOT NULL,
    
    document_type ENUM('Passport', 'Visa', 'Civil ID', 'Labor Card', 'Contract') NOT NULL,
    document_number VARCHAR(50) NOT NULL,
    
    issue_date DATE DEFAULT NULL,
    expiry_date DATE NOT NULL,
    
    -- S3 PRE-SIGNED URL METADATA
    document_key VARCHAR(255) NOT NULL,    -- The unique path in S3
    original_filename VARCHAR(255),        -- e.g., 'my_passport_final.pdf'
    mime_type VARCHAR(100),                -- e.g., 'application/pdf' (Crucial for Presigned URLs)
    file_size_bytes INT UNSIGNED,          -- To show "Size: 1.2 MB" in UI
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (employee_id) REFERENCES employees(id) ON DELETE CASCADE
);
